001    package edu.hawaii.halealohacli.command;
002    
003    /**
004     * Provides the list of available commands and
005     * their parameters.
006     * @author Matthew Mizumoto
007     *
008     */
009    public class Help {
010      
011      
012      /**
013       * Runs the program and
014       * prints out the options for this application.
015       */
016      public static void run()  {
017        System.out.println("Here are the available commands for this system.");
018        System.out.println("current-power [tower | lounge]");
019        System.out.println("Returns the current power in kW for the associated tower or lounge.");
020        System.out.println("daily-energy [tower | lounge] [date]");
021        System.out.println("Returns the energy in kWh used by the tower or lounge " +
022                           "for the specified date (yyyy-mm-dd).");
023        System.out.println("energy-since [tower | lounge] [date]");
024        System.out.println("Returns the energy used since the date (yyyy-mm-dd) to now.");
025        System.out.println("rank-towers [start] [end]");
026        System.out.println("Returns a list in sorted order from least to most energy consumed" +
027                           " between the [start] and [end] date (yyyy-mm-dd)");
028        System.out.println("quit");
029        System.out.println("Terminates execution.");
030        System.out.println("Note: towers are:  Mokihana, Ilima, Lehua, Lokelani");
031        System.out.println("Lounges are the tower names followed by a \"-\" " +
032                           "followed by one of A, B, C, D, E. For example, Mokihana-A.");
033      }
034    
035    }
