001    package org.wattdepot.simpleapp;
002    
003    import java.text.SimpleDateFormat;
004    import java.util.Calendar;
005    import java.util.Date;
006    import java.util.List;
007    import java.util.Locale;
008    import javax.xml.datatype.XMLGregorianCalendar;
009    import org.wattdepot.client.WattDepotClient;
010    import org.wattdepot.resource.sensordata.jaxb.SensorData;
011    import org.wattdepot.resource.source.jaxb.Source;
012    import org.wattdepot.util.tstamp.Tstamp;
013    
014    /**
015     * A simple application that shows basic interaction with a running WattDepot server.
016     * @author Philip Johnson
017     */
018    public class SimpleApplication {
019      
020      /** Holds the WattDepotClient instance used to interact with this server. */
021      private WattDepotClient client;
022      
023      /**
024       * Instantiates this class and ensures that the passed URL identifies a running client. 
025       * @param url A URL that should point to a running WattDepot server.
026       * @throws Exception If the URL does not appear to point to a running WattDepot server.
027       */
028      public SimpleApplication(String url) throws Exception {
029        this.client = new WattDepotClient(url);
030        if (!this.client.isHealthy()) {
031          throw new Exception("WattDepot URL does not point to a server: " + url);
032        }
033      }
034      
035      /**
036       * Returns the latency in seconds associated with the passed sensor data instance. 
037       * @param data The sensor data item.
038       * @return The number of seconds since that data was recorded.
039       */
040      private long getSensorDataLatency(SensorData data) {
041        XMLGregorianCalendar timestamp = data.getTimestamp();
042        long now = Calendar.getInstance(Locale.US).getTimeInMillis();
043        long latency = now - timestamp.toGregorianCalendar().getTimeInMillis();
044        return (latency / 1000);
045      }
046      
047      /**
048       * Get the latest recorded power for a source.
049       * Throws an exception if this data is more than one minute old.
050       * @param source The source of interest.
051       * @return The latest recorded power.
052       * @throws Exception If latency of latest power exceeds 60 seconds. 
053       */
054      public double getCurrentPower (String source) throws Exception {
055        SensorData data = client.getLatestSensorData(source);
056        long latency = getSensorDataLatency(data);
057        if (latency > 60) {
058          throw new Exception("Sensor Data item has latency over 1 minute.");
059        }
060        return data.getPropertyAsDouble("powerConsumed");
061      }
062      
063      /** 
064       * Returns the daily energy associated with the given source on the given day.
065       * @param source  The source. 
066       * @param dayString The day of interest in YYYY-MM-DD format.
067       * @return The amount of energy consumed by that source on that day.
068       * @throws Exception If problems occur obtaining that data from WattDepot.
069       */
070      public double getDailyEnergy(String source, String dayString) throws Exception {
071        Date date = new SimpleDateFormat("yyyy-MM-dd", Locale.US).parse(dayString);
072        XMLGregorianCalendar startOfDay = Tstamp.makeTimestamp(date.getTime());
073        startOfDay.setTime(0,0,0);
074        XMLGregorianCalendar endOfDay = Tstamp.incrementDays(Tstamp.makeTimestamp(date.getTime()), 1);
075        endOfDay.setTime(0, 0, 0);
076        return client.getEnergyConsumed(source, startOfDay, endOfDay, 0);    
077      }
078      
079      /**
080       * Returns the energy consumed for the given source since the start of the given day.
081       * @param source The source of interest.
082       * @param dayString The start day.
083       * @return The total energy consumed in watts since the start day to the latest sensor data. 
084       * @throws Exception If problems occur with WattDepot, or the start day is after the latest
085       * sensor data. 
086       */
087      public double getEnergySince(String source, String dayString) throws Exception {
088        Date date = new SimpleDateFormat("yyyy-MM-dd", Locale.US).parse(dayString);
089        XMLGregorianCalendar startTimestamp = Tstamp.makeTimestamp(date.getTime());
090        startTimestamp.setTime(0,0,0);
091        XMLGregorianCalendar endTimestamp = client.getLatestSensorData(source).getTimestamp();
092        return client.getEnergyConsumed(source, startTimestamp, endTimestamp, 0);    
093      }
094      
095      /**
096       * Returns the WattDepotClient associated with this instance. 
097       * @return The WattDepotClient.
098       */
099      public WattDepotClient getClient() {
100        return this.client;
101      }
102    
103      /**
104       * Illustrate simple interaction with WattDepot.
105       * @param args The first arg should be the URL to a running WattDepot server.
106       * @throws Exception If problems occur communicating with the server.
107       */
108      public static void main(String[] args) throws Exception {
109        SimpleApplication application = new SimpleApplication(args[0]);
110        WattDepotClient client = application.getClient();
111    
112        // Get the list of sources and print them out.
113        System.out.println("WattDepot sources defined for this URL:");
114        List<Source> sources = client.getSources();
115        for (Source source : sources) {
116          System.out.format("Name: %20s      %s%n", source.getName(), source.getDescription());
117        }
118        
119        String sourceName = sources.get(0).getName();
120    
121        System.out.format("Current power for %s is %,.2f kilowatts.%n", 
122            sourceName, (application.getCurrentPower(sourceName) / 1000));
123        
124        String date = "2011-11-01";
125        System.out.format("Daily energy for source %s on %s is %,.2f kilowatts.%n",
126            sourceName, date, (application.getDailyEnergy(sourceName, date) / 1000));
127    
128        System.out.format("Energy since %s for source %s is %,.2f kWh.%n",
129            date, sourceName, (application.getEnergySince(sourceName, date) / 1000));
130    
131    
132      }
133    }
