001    package org.wattdepot.simpleapp;
002    
003    import java.util.Calendar;
004    import java.util.List;
005    import java.util.Locale;
006    import javax.xml.datatype.XMLGregorianCalendar;
007    import org.wattdepot.client.WattDepotClient;
008    import org.wattdepot.resource.sensordata.jaxb.SensorData;
009    import org.wattdepot.resource.source.jaxb.Source;
010    
011    /**
012     * A simple application that shows basic interaction with a running WattDepot server.
013     * @author Philip Johnson
014     */
015    public class SimpleApplication {
016    
017      /**
018       * Illustrate simple interaction with WattDepot.
019       * @param args The first arg should be the URL to a running WattDepot server.
020       * @throws Exception If problems occur communicating with the server.
021       */
022      public static void main(String[] args) throws Exception {
023        String url = args[0];
024        // Create a client.
025        WattDepotClient client = new WattDepotClient(url);
026        // Check to make sure a connection can be made. 
027        // If no connection, then exit right now.
028        if (client.isHealthy()) {
029          System.out.format("Connected successfully to: %s%n", url);
030        }
031        else {
032          System.out.format("Could not connect to: %s%n", url);
033          return;
034        }
035        // Get the list of sources and print them out.
036        System.out.println("WattDepot sources defined for this URL:");
037        List<Source> sources = client.getSources();
038        for (Source source : sources) {
039          System.out.format("Name: %20s      %s%n", source.getName(), source.getDescription());
040        }
041        
042        // Find the last sensor data from the first source in our list.
043        String sourceName = sources.get(0).getName();
044        SensorData data = client.getLatestSensorData(sourceName);
045        
046        // Show the sensor data item
047        System.out.format("Latest energy data from source %s is %s.%n", sourceName, data);
048        
049        // Let's figure out how long ago this energy information occurred.
050        XMLGregorianCalendar timestamp = data.getTimestamp();
051        long now = Calendar.getInstance(Locale.US).getTimeInMillis();
052        long latency = now - timestamp.toGregorianCalendar().getTimeInMillis();
053        System.out.format("This data is %s seconds old.", (latency / 1000));
054      }
055    }
