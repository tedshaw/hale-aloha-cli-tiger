001    package edu.hawaii.halealohacli;
002    
003    import java.util.Scanner;
004    import org.wattdepot.client.WattDepotClient;
005    import edu.hawaii.halealohacli.processor.CommandProcessor;
006    
007    /**
008     * Main class that prompts the user for commands.
009     * @author Matthew Mizumoto
010     *
011     */
012    public class Main {
013      
014      /**
015       * 
016       * @param args no arguments required
017       * @throws Exception if cannot make connection to server
018       */
019      public static void main (String[] args) throws Exception  {
020        String url = "http://server.wattdepot.org:8190/wattdepot/";
021        // Create a client.
022        WattDepotClient client = new WattDepotClient(url);
023        // Check to make sure a connection can be made. 
024        // If no connection, then exit right now.
025        if (client.isHealthy()) {
026          System.out.println("Sucscessfully connected to the Hale Aloha Wattdepot Server");
027        }
028        else {
029          System.out.format("Could not connect to: %s%n", url);
030          return;
031        }
032        
033        Scanner keybd = new Scanner(System.in);
034        String input = keybd.nextLine();
035          while (!("quit".equals(input)))  {
036            CommandProcessor command = new CommandProcessor(input);
037            command.run();
038            input = keybd.nextLine();
039          }
040      }
041    
042    }
