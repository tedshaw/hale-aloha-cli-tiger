001    package edu.hawaii.halealohacli.processor;
002    
003    import javax.xml.datatype.DatatypeConfigurationException;
004    import edu.hawaii.halealohacli.command.Help;
005    
006    /**
007     * Processes the command given by the user
008     * and executes then executes it.
009     * @author mizumoto
010     *
011     */
012    public class CommandProcessor {
013      
014      private String userInput;
015      
016      /**
017       * Constuctor method to access class level methods.
018       * @param userInput Command given by the user.
019       */
020      public CommandProcessor(String userInput)  {
021        this.userInput = userInput;
022      }
023      
024      /**
025       * Chooses module based on the command input.
026       * @param command Command given by the user.
027       */
028      public void chooseModule(String command)  {
029        if ("current-power".equals(command))  {
030          System.out.println("currentPower");
031        }
032        else if ("daily-energy".equals(command))  {
033          System.out.println("dailyEnergy");
034        } 
035        else if ("energy-since".equals(command))  {
036          System.out.println("energySince");
037        } 
038        else if ("rank-towers".equals(command))  {
039          System.out.println("rankTowers");
040        } 
041        else if ("help".equals(command))  {
042          Help.run();
043        } 
044        else  {
045          System.out.println("Not a valid command");
046        }
047      }
048      
049      
050      /**
051       * Runs all error checks on the command and executes it
052       * if command is valid.
053       * @throws DatatypeConfigurationException 
054       */
055      public void run() throws DatatypeConfigurationException  {
056        CommandProcessor input = new CommandProcessor(this.userInput);
057        String[] cmd = this.userInput.split(" ");
058        input.chooseModule(cmd[0]);
059        }
060    }
